**Goal:** 实现需求3 SDK：`linetrack_to_linkinfos(track)` 把 GPS 轨迹字符串经 DB 空间匹配转成有序 `LinkInfo` 列表（含单车道反向 `reverse_coords` 判定）；`linkinfos_to_tracks(linkinfos)` 逆向拼回轨迹字符串。

**Architecture:** 三层。`geometry.py` 为纯函数（解析/格式化/距离/方位角/反向判定），零外部依赖、最易测。`matcher.py` 用 Plan 1 的数据层只读 `road_link_coord` 几何：bbox 粗筛候选 link → 逐 GPS 点最近折线距离匹配（容差内）→ 折叠为有序 link 段。`conversion.py` 暴露 `LinkInfo` dataclass 与 `TrackConverter`（持有 engine + sdk 参数，方法名与需求契约一致）。

**Tech Stack:** Python 3.11、SQLAlchemy 2.0 Core（复用 Plan 1 的 `amap_service/db/schema.py`、`engine`、`migrate`、`repositories`）、标准库 `math`/`dataclasses`、pytest。

**对应 spec：** [docs/superpowers/specs/2026-05-29-amap-service-design.md](../specs/2026-05-29-amap-service-design.md) 第 7 节。**前置依赖：** Plan 1（已合并到 main）。

**关键约定：**
- 坐标顺序统一 **(经度 lng, 纬度 lat)**（高德约定）。
- 轨迹字符串：点内逗号、点间分号，例 `"120.0,31.0;120.001,31.0"`。解析时同时接受全角 `；`/`，`，输出统一用半角。
- 距离采用局部等距投影（equirectangular）近似——路段短、容差 ~30m，足够。
- 反向判定：GPS 子轨迹整体方位角 与 该 link 存储几何整体方位角 的夹角 > `reverse_angle_deg`（默认 90°）→ `reverse_coords=True`。仅当两侧都 ≥2 点时才判定，否则 False。

```python
"""Pure geometry helpers for GPS track ↔ road-link conversion.

Coordinates are (longitude, latitude) tuples in degrees (Amap convention).
No external dependencies, no DB access — fully unit-testable.
"""
import math

EARTH_RADIUS_M = 6371000.0


def parse_track(track: str) -> list[tuple[float, float]]:
    """Parse "lng,lat;lng,lat" into [(lng, lat), ...]. Accepts full-width ；/，."""
    if not track:
        return []
    normalized = track.replace("；", ";").replace("，", ",")
    points: list[tuple[float, float]] = []
    for chunk in normalized.split(";"):
        chunk = chunk.strip()
        if not chunk:
            continue
        lng_s, lat_s = chunk.split(",")
        points.append((float(lng_s), float(lat_s)))
    return points


def format_track(coords: list[tuple[float, float]]) -> str:
    """Inverse of parse_track using half-width separators."""
    return ";".join(f"{lng},{lat}" for lng, lat in coords)


def haversine(p1: tuple[float, float], p2: tuple[float, float]) -> float:
    """Great-circle distance in meters between two (lng, lat) points."""
    lng1, lat1 = p1
    lng2, lat2 = p2
    phi1, phi2 = math.radians(lat1), math.radians(lat2)
    dphi = math.radians(lat2 - lat1)
    dlam = math.radians(lng2 - lng1)
    a = math.sin(dphi / 2) ** 2 + math.cos(phi1) * math.cos(phi2) * math.sin(dlam / 2) ** 2
    return 2 * EARTH_RADIUS_M * math.asin(math.sqrt(a))


def _to_local_xy(p: tuple[float, float], lat0: float) -> tuple[float, float]:
    """Equirectangular projection to local meters, centered at latitude lat0."""
    lng, lat = p
    x = math.radians(lng) * math.cos(math.radians(lat0)) * EARTH_RADIUS_M
    y = math.radians(lat) * EARTH_RADIUS_M
    return x, y


def point_to_segment_distance(
    p: tuple[float, float], a: tuple[float, float], b: tuple[float, float]
) -> float:
    """Shortest distance (meters) from point p to segment a-b (planar approx)."""
    lat0 = a[1]
    px, py = _to_local_xy(p, lat0)
    ax, ay = _to_local_xy(a, lat0)
    bx, by = _to_local_xy(b, lat0)
    dx, dy = bx - ax, by - ay
    if dx == 0.0 and dy == 0.0:
        return math.hypot(px - ax, py - ay)
    t = ((px - ax) * dx + (py - ay) * dy) / (dx * dx + dy * dy)
    t = max(0.0, min(1.0, t))
    cx, cy = ax + t * dx, ay + t * dy
    return math.hypot(px - cx, py - cy)


def bearing(a: tuple[float, float], b: tuple[float, float]) -> float:
    """Initial bearing in degrees [0, 360) from point a to point b."""
    lng1, lat1 = math.radians(a[0]), math.radians(a[1])
    lng2, lat2 = math.radians(b[0]), math.radians(b[1])
    dlon = lng2 - lng1
    x = math.sin(dlon) * math.cos(lat2)
    y = math.cos(lat1) * math.sin(lat2) - math.sin(lat1) * math.cos(lat2) * math.cos(dlon)
    return (math.degrees(math.atan2(x, y)) + 360.0) % 360.0


def angle_diff(b1: float, b2: float) -> float:
    """Smallest absolute difference between two bearings, in [0, 180]."""
    d = abs(b1 - b2) % 360.0
    return d if d <= 180.0 else 360.0 - d


def is_reverse(track_bearing: float, link_bearing: float, threshold_deg: float = 90.0) -> bool:
    """True when the track heads opposite the stored link geometry."""
    return angle_diff(track_bearing, link_bearing) > threshold_deg


def polyline_bearing(coords: list[tuple[float, float]]) -> float:
    """Overall bearing from first to last coordinate; 0.0 if fewer than 2 points."""
    if len(coords) < 2:
        return 0.0
    return bearing(coords[0], coords[-1])
```

`amap_service/sdk/matcher.py`:
```python
"""DB-backed map matching: GPS track → ordered road-link sequence.

Reads road_link_coord geometry only; does not care how it was landed.
"""
from dataclasses import dataclass, field

from sqlalchemy import Engine, select

from amap_service.db.schema import road_link_coord
from amap_service.sdk import geometry

# Rough degrees-per-meter for bbox expansion (latitude; longitude is tighter but
# over-expanding the candidate box is harmless — only correctness of the filter matters).
_DEG_PER_M = 1.0 / 111320.0


@dataclass
class MatchedLink:
    link_id: int
    gps_coords: list = field(default_factory=list)  # GPS points assigned to this link, in order


class LinkMatcher:
    def __init__(self, engine: Engine, tolerance_m: float = 30.0):
        self.engine = engine
        self.tolerance_m = tolerance_m

    def match_track(self, coords: list[tuple[float, float]]) -> list[MatchedLink]:
        if not coords:
            return []
        geoms = self._load_candidate_geometries(coords)
        if not geoms:
            return []

        result: list[MatchedLink] = []
        for point in coords:
            best_id, best_dist = None, None
            for link_id, polyline in geoms.items():
                dist = self._point_to_polyline(point, polyline)
                if best_dist is None or dist < best_dist:
                    best_dist, best_id = dist, link_id
            if best_dist is None or best_dist > self.tolerance_m:
                continue  # drop points beyond tolerance
            if result and result[-1].link_id == best_id:
                result[-1].gps_coords.append(point)
            else:
                result.append(MatchedLink(best_id, [point]))
        return result

    def _load_candidate_geometries(self, coords) -> dict:
        lngs = [c[0] for c in coords]
        lats = [c[1] for c in coords]
        margin = self.tolerance_m * _DEG_PER_M * 2.0
        min_lng, max_lng = min(lngs) - margin, max(lngs) + margin
        min_lat, max_lat = min(lats) - margin, max(lats) + margin

        with self.engine.connect() as conn:
            candidate_ids = set(
                conn.execute(
                    select(road_link_coord.c.link_id)
                    .where(road_link_coord.c.longitude.between(min_lng, max_lng))
                    .where(road_link_coord.c.latitude.between(min_lat, max_lat))
                    .distinct()
                ).scalars().all()
            )
            if not candidate_ids:
                return {}
            rows = conn.execute(
                select(
                    road_link_coord.c.link_id,
                    road_link_coord.c.longitude,
                    road_link_coord.c.latitude,
                )
                .where(road_link_coord.c.link_id.in_(candidate_ids))
                .order_by(road_link_coord.c.link_id, road_link_coord.c.seq)
            ).all()

        geoms: dict[int, list] = {}
        for row in rows:
            geoms.setdefault(row.link_id, []).append((row.longitude, row.latitude))
        return geoms

    @staticmethod
    def _point_to_polyline(point, polyline) -> float:
        if len(polyline) == 1:
            return geometry.haversine(point, polyline[0])
        return min(
            geometry.point_to_segment_distance(point, polyline[i], polyline[i + 1])
            for i in range(len(polyline) - 1)
        )
```

```python
"""Public SDK: bidirectional GPS-track ↔ ordered road-link conversion."""
from dataclasses import dataclass

from sqlalchemy import Engine, select

from amap_service.db.schema import road_link_coord
from amap_service.sdk import geometry
from amap_service.sdk.matcher import LinkMatcher


@dataclass
class LinkInfo:
    link_id: int
    reverse_coords: bool


class TrackConverter:
    def __init__(self, engine: Engine, tolerance_m: float = 30.0, reverse_angle_deg: float = 90.0):
        self.engine = engine
        self.reverse_angle_deg = reverse_angle_deg
        self.matcher = LinkMatcher(engine, tolerance_m)

    def linetrack_to_linkinfos(self, track: str) -> list[LinkInfo]:
        """GPS track string → ordered LinkInfo list with reverse_coords flags."""
        coords = geometry.parse_track(track)
        matched = self.matcher.match_track(coords)
        if not matched:
            return []
        geoms = self._load_geometries([m.link_id for m in matched])
        infos: list[LinkInfo] = []
        for m in matched:
            polyline = geoms.get(m.link_id, [])
            reverse = False
            if len(m.gps_coords) >= 2 and len(polyline) >= 2:
                gps_bearing = geometry.polyline_bearing(m.gps_coords)
                link_bearing = geometry.polyline_bearing(polyline)
                reverse = geometry.is_reverse(gps_bearing, link_bearing, self.reverse_angle_deg)
            infos.append(LinkInfo(link_id=m.link_id, reverse_coords=reverse))
        return infos

    def linkinfos_to_tracks(self, linkinfos: list[LinkInfo]) -> str:
        """Ordered LinkInfo list → concatenated GPS track string (seam endpoints de-duped)."""
        if not linkinfos:
            return ""
        geoms = self._load_geometries([li.link_id for li in linkinfos])
        merged: list[tuple[float, float]] = []
        for li in linkinfos:
            polyline = list(geoms.get(li.link_id, []))
            if li.reverse_coords:
                polyline = polyline[::-1]
            for point in polyline:
                if merged and merged[-1] == point:
                    continue  # drop duplicated seam endpoint
                merged.append(point)
        return geometry.format_track(merged)

    def _load_geometries(self, link_ids: list[int]) -> dict:
        if not link_ids:
            return {}
        with self.engine.connect() as conn:
            rows = conn.execute(
                select(
                    road_link_coord.c.link_id,
                    road_link_coord.c.longitude,
                    road_link_coord.c.latitude,
                )
                .where(road_link_coord.c.link_id.in_(set(link_ids)))
                .order_by(road_link_coord.c.link_id, road_link_coord.c.seq)
            ).all()
        geoms: dict[int, list] = {}
        for row in rows:
            geoms.setdefault(row.link_id, []).append((row.longitude, row.latitude))
        return geoms
```

## 已知简化（记录，不阻塞）

- 逐点最近匹配 + 连续折叠：对环线/折返（A→B→A）会输出重复 link，这是期望行为；但对噪声极大的真实 GPS 可能需要更强的 HMM 匹配——留待需求2 真实数据验证后按需增强。
- **反向判定用 `polyline_bearing`（首点→末点弦向）**：对短、走向单调的路段足够准确（需求2 公交按 link 分段后子轨迹通常如此）。但对单 link 内含发卡弯/U 形的几何，弦向可能误判方向。注意「按位移向量求和」与首末弦向数学等价（位移求和望远镜式抵消为 `末-首`），并非真正改进；真正更稳健的做法是「单位方向向量平均」或 HMM，留待需求2 真实数据出现误判后再做。
- **SdkConfig 接线**：`config.sdk`（`match_tolerance_m`/`reverse_angle_deg`）需由调用方（Plan 5 需求2 流水线）桥接到 `TrackConverter(...)` 构造参数；SDK 作为库不直接读全局配置。`去抖(dedup_jitter_m)` 当前仅以「连续相同 link 折叠」近似，未单独使用该参数。
- **接缝去重用精确相等**：相邻 link 共享拓扑节点时端点通常为同一 float、可正确去重；若上游出现独立舍入的端点，则去重静默失效（留作真实数据验证项，必要时改用距离阈值去重）。
- bbox 候选检索在大库（911k 坐标）上依赖 `road_link_coord(longitude/latitude)` 的范围扫描；Plan 1 已建 `link_id` 索引但未建经纬度索引。若需求2 批量匹配出现性能问题，再在 migrate 增加 `(longitude, latitude)` 复合索引（届时单列一个 task）。
